# M5 单文件前端实施计划

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** 用单文件 HTML + 原生 JS + Pico.css CDN 实现完整前端：人员管理（CRUD）+ 实时识别画面（WebSocket + Canvas 叠加框）。

**Architecture:** 一个 `index.html`，内嵌全部 CSS 和 JS。两个 Tab 切换"人员管理"和"实时识别"。前端不写打包工具、不引框架——一个文件直接 `app.mount("/", StaticFiles(...))` 即可托管。

**Tech Stack:** 原生 HTML5 + JavaScript（ES2022, fetch / WebSocket / Canvas API） + [Pico.css](https://picocss.com/) CDN（极简语义化 CSS 框架，类似 Bootstrap 但只有 ~10KB）。

- [ ] **Step 1: 在 `<script>` 中加 tab 切换 + 列表渲染**

```javascript
"use strict";

// ===== 工具函数 =====
const toastEl = document.getElementById("toast");
let toastTimer;
function showToast(message, type = "success", duration = 3000) {
  toastEl.textContent = message;
  toastEl.className = `toast ${type}`;
  clearTimeout(toastTimer);
  toastTimer = setTimeout(() => toastEl.classList.add("hidden"), duration);
}

function escapeHtml(str) {
  const div = document.createElement("div");
  div.textContent = String(str);
  return div.innerHTML;
}

async function apiFetch(path, options = {}) {
  const resp = await fetch(path, options);
  if (!resp.ok) {
    let detail = `HTTP ${resp.status}`;
    try { const body = await resp.json(); detail = body.detail || detail; } catch (_) {}
    throw new Error(detail);
  }
  if (resp.status === 204) return null;
  return resp.json();
}

// ===== Tab 切换 =====
const tabPersons = document.getElementById("tab-persons");
const tabStream  = document.getElementById("tab-stream");
const pagePersons = document.getElementById("page-persons");
const pageStream  = document.getElementById("page-stream");

function switchTab(active) {
  if (active === "persons") {
    pagePersons.hidden = false; pageStream.hidden = true;
    tabPersons.classList.add("tab-active"); tabStream.classList.remove("tab-active");
    loadPersons();
  } else {
    pagePersons.hidden = true; pageStream.hidden = false;
    tabStream.classList.add("tab-active"); tabPersons.classList.remove("tab-active");
  }
}
tabPersons.addEventListener("click", e => { e.preventDefault(); switchTab("persons"); });
tabStream .addEventListener("click", e => { e.preventDefault(); switchTab("stream");  });

// ===== 人员管理 =====
const personsTbody = document.getElementById("persons-tbody");
const registerForm = document.getElementById("register-form");

async function loadPersons() {
  try {
    const persons = await apiFetch("/api/persons");
    personsTbody.innerHTML = "";
    if (!persons.length) {
      personsTbody.innerHTML = `<tr><td colspan="4"><em>暂无人员</em></td></tr>`;
      return;
    }
    for (const p of persons) {
      const tr = document.createElement("tr");
      tr.innerHTML = `
        <td><code>${escapeHtml(p.person_id)}</code></td>
        <td>${escapeHtml(p.display_name)}</td>
        <td>${p.template_count}</td>
        <td><button class="outline secondary" data-action="delete" data-id="${escapeHtml(p.person_id)}">删除</button></td>
      `;
      personsTbody.appendChild(tr);
    }
  } catch (err) {
    showToast(`加载失败: ${err.message}`, "error");
  }
}
```

// 事件代理：删除
personsTbody.addEventListener("click", async (e) => {
  const btn = e.target.closest("button[data-action=delete]");
  if (!btn) return;
  const id = btn.dataset.id;
  if (!confirm(`确定删除 ${id} 吗？此操作不可撤销。`)) return;
  btn.setAttribute("aria-busy", "true"); btn.disabled = true;

    showToast(`已删除: ${id}`, "success");
    loadPersons();
  } catch (err) {
    showToast(`删除失败: ${err.message}`, "error");
    btn.removeAttribute("aria-busy"); btn.disabled = false;

});

### Task 5: 实时识别 WebSocket 连接

**Files:**
- Modify: `src/face_recognition/api/static/index.html`（追加 JS）

**关键概念**：

- **WebSocket API**：浏览器原生 `new WebSocket(url)`，事件 `open / message / error / close`。`message` 事件的 `event.data` 类型取决于服务端发的是文本还是二进制。
- **Blob**：浏览器中表示二进制数据的对象。WebSocket 默认 `binaryType = "blob"`，也可改为 `"arraybuffer"`。这里我们直接把 Blob 喂给 `createImageBitmap` 解码。
- **Canvas 2D Context**：`canvas.getContext("2d")`，可以 `drawImage` 把 `ImageBitmap` / `Image` 画上去。
- **createImageBitmap(blob)**：异步把 Blob 解码成 `ImageBitmap`(GPU 友好的位图对象),浏览器在后台线程解码不阻塞主线程。比"`new Image()` + `URL.createObjectURL` + `onload`"链路更现代,且不需要手动 `revokeObjectURL` 释放。**用完调 `bitmap.close()` 释放显存**。

- [ ] **Step 1: 加 WebSocket 客户端**

```javascript
// ===== 实时识别 WebSocket =====
const canvas = document.getElementById("video-canvas");
const ctx = canvas.getContext("2d");
const btnStart = document.getElementById("btn-start-stream");
const btnStop  = document.getElementById("btn-stop-stream");
const tracksTbody = document.getElementById("tracks-tbody");

let ws = null;

function startStream() {
  if (ws) return;
  const proto = location.protocol === "https:" ? "wss" : "ws";
  ws = new WebSocket(`${proto}://${location.host}/ws/stream`);
  ws.binaryType = "blob";

  ws.onopen = () => {
    showToast("已连接，识别中...", "success");
    btnStart.disabled = true; btnStop.disabled = false;
  };

  ws.onmessage = async (event) => {
    if (event.data instanceof Blob) {
      // 二进制 JPEG 帧
      const bitmap = await createImageBitmap(event.data);
      if (canvas.width !== bitmap.width || canvas.height !== bitmap.height) {
        canvas.width = bitmap.width; canvas.height = bitmap.height;
      }
      ctx.drawImage(bitmap, 0, 0);
      bitmap.close();
    } else {
      // JSON 元数据
      let meta;
      try { meta = JSON.parse(event.data); } catch (_) { return; }
      if (meta.error) { showToast(`服务端错误: ${meta.error}`, "error"); stopStream(); return; }
      renderTracks(meta.tracks || []);
    }
  };

  ws.onerror = () => showToast("WebSocket 连接错误", "error");
  ws.onclose = () => { ws = null; btnStart.disabled = false; btnStop.disabled = true; };
}

function stopStream() { if (ws) ws.close(); }

function renderTracks(tracks) {
  tracksTbody.innerHTML = "";
  if (!tracks.length) {
    tracksTbody.innerHTML = `<tr><td colspan="3"><em>未检测到人脸</em></td></tr>`;
    return;
  }
  for (const t of tracks) {
    const tr = document.createElement("tr");
    tr.innerHTML = `
      <td>${t.track_id}</td>
      <td><span class="${t.identity ? 'identity-known' : 'identity-unknown'}">${t.identity ? escapeHtml(t.identity) : '未知'}</span></td>
      <td>${t.similarity.toFixed(3)}</td>
    `;
    tracksTbody.appendChild(tr);
  }
}

btnStart.addEventListener("click", startStream);
btnStop .addEventListener("click", stopStream);

// 首屏加载
loadPersons();
```
